{

}

第2题：整除问题

#include <stdio.h>
int main()
{
    int a,b;
    scanf("%d%d",&a,&b);
    if(a % b == 0)
        printf("YES");
    else
        printf("NO");
}

第3题：判断质数

#include <stdio.h>
int main()
{
    int a,i;
    scanf("%d",&a);
    for(i = 2;i < a;i++)
    {
        if(a % i == 0)
        {
            printf("NO");
            return 0;
        }
        else
            continue;
    }
    if(i == a)
        printf("YES");
}
